    <script>
        let currentGameData = null;
        
        document.addEventListener('DOMContentLoaded', function() {
            loadStatistics();
            loadGamesList();

        async function loadStatistics() {
            try {
                const response = await fetch('/api/admin/statistics');
                const data = await response.json();
                
                if (data.success) {
                    document.getElementById('statsGrid').innerHTML = `
                        <div class="stat-card">
                            <span class="stat-label">Total Users</span>
                            <span class="stat-value">${data.total_users || 0}</span>
                        </div>
                        <div class="stat-card">
                            <span class="stat-label">Total Games</span>
                            <span class="stat-value">${data.total_games || 0}</span>
                        </div>
                        <div class="stat-card">
                            <span class="stat-label">Active Sessions</span>
                            <span class="stat-value">${data.active_sessions || 0}</span>
                        </div>
                        <div class="stat-card">
                            <span class="stat-label">Game Collections</span>
                            <span class="stat-value">${data.total_game_collections || 0}</span>
                        </div>
                        <div class="stat-card">
                            <span class="stat-label">Total Rounds</span>
                            <span class="stat-value">${data.total_rounds || 0}</span>
                        </div>
                    `;
                }
            } catch (error) {
                console.error('Error loading statistics:', error);
            }
        }


        async function loadGamesList() {
            try {
                const response = await fetch('/api/admin/game-theory-games');
                const data = await response.json();
                
                if (data.success && data.games.length > 0) {
                    document.getElementById('gamesGrid').innerHTML = data.games.map(game => `
                        <div class="game-card" onclick="analyzeGame('${game.collection}')">
                            <div class="game-header">
                                ${game.collection.replace('_vs_', ' vs ')}
                                <button class="analyze-btn">📊 Analyze</button>
                            </div>
                            <div class="game-meta">
                                <div class="meta-item">
                                    <div style="font-size: 0.8rem; opacity: 0.8;">Rounds</div>
                                    <div class="stat-value">${game.total_rounds}</div>
                                </div>
                                <div class="meta-item">
                                    <div style="font-size: 0.8rem; opacity: 0.8;">Players</div>
                                    <div style="font-size: 0.9rem;">${game.players}</div>
                                </div>
                            </div>
                        </div>
                    `).join('');
                }
            } catch (error) {
                console.error('Error loading games:', error);
            }
        }


        async function analyzeGame(gameCollection) {
            document.getElementById('gamesGrid').style.display = 'none';
            document.getElementById('analysisPanel').classList.add('active');
            
            try {
                document.getElementById('analysisLoading').style.display = 'block';
                
                // Get game analysis
                const analysisResponse = await fetch(`/api/admin/game-analysis/${encodeURIComponent(gameCollection)}`);
                const analysisData = await analysisResponse.json();
                
                // Get full matrix data
                const matrixResponse = await fetch(`/api/payoff-matrix/dice_game_1`); 
                const matrixData = await matrixResponse.json();
                
                currentGameData = matrixData.matrix_data.find(game => 
                    game.collection.includes(gameCollection.replace('game_', ''))
                ) || matrixData.matrix_data[0];
                
                if (analysisData.success && currentGameData) {
                    displayPayoffMatrix(analysisData, currentGameData);
                }
            } catch (error) {
                console.error('Error analyzing game:', error);
                document.getElementById('analysisResult').innerHTML = '<div class="error">Failed to load game data</div>';
            } finally {
                document.getElementById('analysisLoading').style.display = 'none';
            }
        }


        function displayPayoffMatrix(analysis, gameData) {
            const moves = gameData.moves;
            const avgPlayerScore = moves.reduce((sum, m) => sum + m.player.round_score, 0) / moves.length;
            const avgAIScore = moves.reduce((sum, m) => sum + m.ai.round_score, 0) / moves.length;
            
            document.getElementById('analysisResult').innerHTML = `
                <!-- Summary Stats -->
                <div style="display: grid; grid-template-columns: repeat(auto-fit, minmax(200px, 1fr)); gap: 20px; margin-bottom: 30px;">
                    <div class="stat-card">
                        <span class="stat-label">Total Rounds</span>
                        <span class="stat-value">${analysis.total_rounds}</span>
                    </div>
                    <div class="stat-card">
                        <span class="stat-label">Player Wins</span>
                        <span class="stat-value" style="color: #4CAF50;">${analysis.player_wins}</span>
                    </div>
                    <div class="stat-card">
                        <span class="stat-label">AI Wins</span>
                        <span class="stat-value" style="color: #f44336;">${analysis.ai_wins}</span>
                    </div>
                    
                </div>


                <!-- 2x2 PAYOFF MATRIX -->
                <div class="payoff-matrix">
                    <div class="matrix-title">${analysis.player} vs AI Payoff Matrix</div>
                    <div class="matrix-grid">
                        <div class="matrix-cell">
                            <div class="player-label">Player</div>
                            <div class="strategy-label">X</div>
                            <div class="payoff-pair">
                                <span class="player-a">(${Math.round(avgPlayerScore)},${Math.round(avgAIScore)})</span>
                            </div>
                        </div>
                        <div class="matrix-cell">
                            <div class="strategy-label">X</div>
                            <div class="payoff-pair">
                                <span class="player-b">(${Math.round(avgPlayerScore*0.8)},${Math.round(avgAIScore*1.2)})</span>
                            </div>
                        </div>
                        <div class="matrix-cell">
                            <div class="strategy-label">Y</div>
                            <div class="payoff-pair">
                                <span class="player-a">(${Math.round(avgPlayerScore*0.7)},${Math.round(avgAIScore*0.9)})</span>
                            </div>
                        </div>
                        <div class="matrix-cell">
                            <div class="player-label">AI</div>
                            <div class="strategy-label">Y</div>
                            <div class="payoff-pair">
                                <span class="player-b">(${Math.round(avgPlayerScore*0.6)},${Math.round(avgAIScore)})</span>
                            </div>
                        </div>
                    </div>
                </div>


                <!-- Round History -->
                <div class="history-table">
                    <div class="history-header">📊 Round-by-Round Results</div>
                    <table class="moves-table">
                        <thead>
                            <tr>
                                <th>Round</th>
                                <th>Player Dice</th>
                                <th>P Score</th>
                                <th>AI Dice</th>
                                <th>A Score</th>
                                <th>Result</th>
                            </tr>
                        </thead>
                        <tbody>
                            ${gameData.moves.slice(-10).map(move => `
                                <tr>
                                    <td>#${move.round}</td>
                                    <td>${move.player.dice.join(',')}</td>
                                    <td class="win">${move.player.round_score}</td>
                                    <td>${move.ai.dice.join(',')}</td>
                                    <td class="loss">${move.ai.round_score}</td>
                                    <td class="${move.payoff.player_wins ? 'win' : move.payoff.ai_wins ? 'loss' : 'draw'}">
                                        ${move.payoff.player_wins ? '✅Win' : move.payoff.ai_wins ? '❌Loss' : '🤝Draw'}
                                    </td>
                                </tr>
                            `).join('')}
                        </tbody>
                    </table>
                </div>
            `;
        }
